120 def retrofit_wall(self, year_of_retrofit, material=None): 

121 """Retrofits wall to German refurbishment standards. 

122 

123 This function adds an additional layer of insulation and sets the 

124 thickness of the layer according to the retrofit standard in the 

125 year of refurbishment. Refurbishment year must be newer then 1977 

126 

127 Note: To Calculate thickness and U-Value, the standard TEASER 

128 coefficients for outer and inner heat transfer are used. 

129 

130 The used Standards are namely the Waermeschutzverordnung (WSVO) and 

131 Energieeinsparverordnung (EnEv) 

132 

133 Parameters 

134 ---------- 

135 material : string 

136 Type of material, that is used for insulation 

137 year_of_retrofit : int 

138 Year of the retrofit of the wall/building 

139 

140 """ 

141 material, year_of_retrofit = self.initialize_retrofit( 

142 material, year_of_retrofit) 

143 

144 calc_u = None 

145 

146 if 1977 <= year_of_retrofit <= 1981: 

147 calc_u = 0.8 

148 elif 1982 <= year_of_retrofit <= 1994: 

149 calc_u = 0.7 

150 elif 1995 <= year_of_retrofit <= 2001: 

151 calc_u = 0.5 

152 elif 2002 <= year_of_retrofit <= 2008: 

153 calc_u = 0.4 

154 elif 2009 <= year_of_retrofit <= 2013: 

155 calc_u = 0.3 

156 elif year_of_retrofit >= 2014: 

157 calc_u = 0.3 

158 

159 self.set_insulation(material, calc_u, year_of_retrofit)
