1# # Example 2: Export Modelica models for AixLib library using TEASER API 

2 

3# This module contains an example how to export buildings from a TEASER 

4# project to ready-to-run simulation models for Modelica library AixLib. These 

5# models will only simulate using Dymola, the reason for this are state 

6# machines that are used in one AixLib specific AHU model. 

7# You can run this example using the [jupyter-notebook](https://mybinder.org/v2/gh/RWTH-EBC/TEASER/main?labpath=docs%2Fjupyter_notebooks) 

8 

9import teaser.examples.e1_generate_archetype as e1 

10import teaser.logic.utilities as utilities 

11import os 

12 

13 

14def example_export_aixlib(): 

15 """This function demonstrates the export to Modelica library AixLib using 

16 the API function of TEASER""" 

17 

18 # In e1_generate_archetype we created a Project with three archetype 

19 # buildings to get this Project we rerun this example 

20 

21 prj = e1.example_generate_archetype() 

22 

23 # To make sure the export is using the desired parameters you should 

24 # always set model settings in the Project. 

25 # Project().used_library_calc specifies the used Modelica library 

26 # Project().number_of_elements_calc sets the models order 

27 # For more information on models we'd like to refer you to the docs. By 

28 # default TEASER uses a weather file provided in 

29 # teaser.data.input.inputdata.weatherdata. You can use your own weather 

30 # file by setting Project().weather_file_path. However we will use default 

31 # weather file. 

32 # Be careful: Dymola does not like whitespaces in names and filenames, 

33 # thus we will delete them anyway in TEASER. 

34 

35 # for CI testing purpose we set the reference result folder 

36 

37 prj.dir_reference_results = utilities.get_full_path( 

38 os.path.join( 

39 "examples", 

40 "examplefiles", 

41 "ReferenceResults", 

42 "Dymola")) 

43 

44 print(prj.dir_reference_results) 

45 

46 prj.used_library_calc = 'AixLib' 

47 prj.number_of_elements_calc = 2 

48 prj.weather_file_path = utilities.get_full_path( 

49 os.path.join( 

50 "data", 

51 "input", 

52 "inputdata", 

53 "weatherdata", 

54 "DEU_BW_Mannheim_107290_TRY2010_12_Jahr_BBSR.mos")) 

55 

56 # To make sure the parameters are calculated correctly we recommend to 

57 # run calc_all_buildings() function 

58 

59 prj.calc_all_buildings() 

60 

61 # To export the ready-to-run models simply call Project.export_aixlib(). 

62 # You can specify the path, where the model files should be saved. 

63 # None means, that the default path in your home directory 

64 # will be used. If you only want to export one specific building, you can 

65 # pass over the internal_id of that building and only this model will be 

66 # exported. In this case we want to export all buildings to our home 

67 # directory, thus we are passing over None for both parameters. 

68 

69 # We might want not have all data stored in our result file. By defining 

70 # export_vars as following we can specify which results we want to store 

71 # and define a collection name under which these results are stored. This 

72 # feature only works with Dymola. 

73 

74 export_vars = { 

75 "HeatingDemands": ["*multizone.PHeater*", "*multizone.PHeatAHU"], 

76 "CoolingDemands": ["*multizone.PCooler*", "*multizone.PCoolAHU"], 

77 "Temperatures": ["*multizone.TAir*", "*multizone.TRad*"] 

78 } 

79 

80 path = prj.export_aixlib( 

81 internal_id=None, 

82 path=None, 

83 report=True, 

84 export_vars=export_vars 

85 ) 

86 

87 return path 

88 

89 

90if __name__ == '__main__': 

91 

92 example_export_aixlib() 

93 

94 print("Example 2: That's it! :)")
