13class Pool(object): 

14 """Pool Class 

15 

16 This class holds information for a pool. 

17 This class is very AixLib specific and holds some parameters that are only 

18 applicable to AixLib pool model. 

19 

20 Parameters 

21 ---------- 

22 parent : ThermalZone() 

23 The parent class of this object, the ThermalZone the pool belongs to. 

24 Allows for better control of hierarchical structures. 

25 Default is None. 

26 

27 Attributes 

28 ---------- 

29 pool_type : str 

30 pool type for the calulation of attributes, currently the following are supported: 

31 'Swimmer_pool', 'Nonswimmer_pool', 'Multipurpose_pool', 'Child_pool', 'Diving_pool','Freeform_pool' 

32 name : str 

33 Individual name. 

34 internal_id : float 

35 Random id for the distinction between different pools. 

36 area : float 

37 Pool area in m2 

38 width : float 

39 Pool width in m. 

40 length : float 

41 Pool length in m. 

42 perimeter : float 

43 Perimeter of pool in m 

44 volume : float 

45 Pool volume in m3. 

46 depth : float 

47 Average pool depth in m. 

48 temperature : float 

49 Pool temperature in K. 

50 temperature_air : float 

51 Air temperature in K. 

52 filter_type : str 

53 Information on used filter type within the water treatment system. 

54 Possible input: '"Activated carbon filter with ozone', 'Two-layer filter with ozone', 

55 'Open quick filter', 'Closed quick filter', 'Closed sorption filter', 'Open suction filter', 

56 '"Quantozone filter', 'Quartz gravel filter', 'Two-layer filter' 

57 filter_combination : str 

58 Information on used filter combination within the water treatment system. 

59 Possible input: 'without_ozone', 'with_ozone', 'with_bromine', 'with_ultrafiltration' 

60 water_type : str 

61 Information on water usage within the pool. 

62 Possible inputs: 'Fresh water" or any other 

63 volume_flow : float 

64 Volume flow of water treatment according to DIN 19643-1 in m3/h" 

65 night_set_back_volume_flow : Boolean 

66 If true, a night set back mode for the volume flow exists 

67 volume_flow_night : float 

68 Volume flow of water treatment at night according to DIN 19643-1 in m3/h 

69 use_partial_load : Boolean 

70 If true, the volume flow of the water treatment is reduced at night. 

71 volume_storage : float 

72 Volume of the water storage within the water treatment system according to DIN 19643-1 in m3 

73 beta_in_use : float 

74 Factor for the evaporation within a used pool according to VDI 2089 

75 use_ideal_heat_exchanger : Boolean 

76 If true, the exported AixLib model includes an ideally heated pool. 

77 dp_heat_exchanger : float 

78 pressure loss of heat exchanger in Pa, only needed if use_ideal_heat_exchanger = False. 

79 use_heat_recovery : Boolean 

80 If true, heat recovery is considered for waste water. 

81 eta_heat_recovery : float 

82 Efficiency for heat recovery. 

83 use_pool_cover: Boolean 

84 If true, pool has a pool cover, which is used during night. 

85 use_wave_pool : Boolean 

86 If true, pool is realised with a wave machine, which is run regular. 

87 wave_height : float 

88 Hight of produced waves in m. 

89 wave_width: float 

90 Width of produced waves in m. 

91 wave_period : float 

92 Length of wave cycling period in sec (wave production and break). 

93 wave_period_start : float 

94 Start time of first wave cycle in sec. 

95 wave_share_period : float 

96 Share of the time with wave generation within the cycling period. 

97 use_water_recycling : Boolean 

98 If true, water recycling unit within the water treatment system 

99 x_recycling : float 

100 Rate of water recycling 

101 num_visitors : float 

102 Average number of visitors per hour in 1/h. 

103 num_filter_rinses : float 

104 Required number of filter rinses per week according to DIN 19643-1 in 1/week 

105 m_flow_waste_water : float 

106 Water exchange due to filter rinses per week or people in the pool, DIN 19643-1 in kg/s 

107 area_pool_wall_inner : float 

108 Area of pool walls which is connected to inner rooms in m2 (inner pool walls). 

109 area_pool_wall_exterior : float 

110 Area of pool walls which is connected to the ground in m2 (pool wall with earth contact). 

111 area_pool_floor_inner : float 

112 Area of pool floor which is connected to inner rooms in m2 (inner pool floor). 

113 area_pool_floor_exterior : float 

114 Area of pool floor which is connected to the ground in m2 (pool floor with earth contact). 

115 h_con_water_horizontal : float 

116 Mean value for the heat transfer coefficient of free convection on horizontal pool floors in W/(m2*K). 

117 h_con_water_vertical : float 

118 Mean value for the heat transfer coefficient of free convection on vertical pool walls in W/(m2*K). 

119 pool_wall_construction_type : str 

120 Construction type of pool wall, supported construction types: 'concrete_with_insulation', 'stainless_steel' 

121 m_flow_evap_pool_used : float 

122 Evaporation mass flow of used pool with maximal occupancy in kg/h 

123 """

191 def calc_pool_parameters(self): 

192 # Typical depths for pools according to DIN 19643 and VDI 2089 

193 if self.depth is None: 

194 if self.pool_type == 'Swimmer_pool': 

195 self.depth = 2.5 

196 elif self.pool_type == 'Nonswimmer_pool': 

197 self.depth = 0.975 

198 elif self.pool_type == 'Child_pool': 

199 self.depth = 0.6 

200 elif self.pool_type == 'Multipurpose_pool': 

201 self.depth = 1.35 

202 elif self.pool_type == 'Diving_pool': 

203 self.depth = 3.8 

204 elif self.pool_type == 'Freeform_pool': 

205 self.depth = 0.75 

206 else: 

207 raise ValueError(f"Unknown pool type: {self.pool_type}. Only the following pool types are supported: " \ 

208 f"'Swimmer_pool', 'Nonswimmer_pool', 'Multipurpose_pool', 'Child_pool'," \ 

209 f" 'Multipurpose_pool', 'Diving_pool', 'Freeform_pool'") 

210 

211 # Typical temperature for pools according to DIN 19643 and VDI 2089 

212 if self.temperature is None: 

213 if self.pool_type == 'Swimmer_pool': 

214 self.temperature = 301.15 

215 elif self.pool_type == 'Nonswimmer_pool': 

216 self.temperature = 303.15 

217 elif self.pool_type == 'Child_pool': 

218 self.temperature = 305.15 

219 elif self.pool_type == 'Multipurpose_pool': 

220 self.temperature = 301.15 

221 elif self.pool_type == 'Diving_pool': 

222 self.temperature = 301.15 

223 elif self.pool_type == 'Freeform_pool': 

224 self.temperature = 303.15 

225 else: 

226 raise ValueError(f"Unknown pool type: {self.pool_type}. Only the following pool types are supported: " \ 

227 f"'Swimmer_pool', 'Nonswimmer_pool', 'Multipurpose_pool', 'Child_pool'," \ 

228 f" 'Multipurpose_pool', 'Diving_pool', 'Freeform_pool'") 

229 

230 

231 if self.volume is None: 

232 self.volume = self.area * self.depth 

233 

234 

235 # Calculation of the nominal volume_flow according to DIN 19643-1 

236 # filter_combination 

237 if self.filter_combination is None: 

238 self.filter_combination = 'without_ozone' 

239 

240 # help parameter k 

241 if self.filter_combination == 'without_ozone' or \ 

242 self.filter_combination == 'with_bromine': 

243 k = 0.5 

244 elif self.filter_combination == 'with_ozone': 

245 k = 0.6 

246 elif self.filter_combination == 'with_ultrafiltration': 

247 k = 1 

248 else: 

249 raise ValueError(f"Unknown filter combination: {self.filter_combination}. " 

250 f"Only the following filter combinations are supported: " 

251 f"'without_ozone', 'with_ozone', 'with_bromine', 'with_ultrafiltration' ") 

252 

253 # help parameters m, a, n ; nominal load N 

254 if self.pool_type == 'Swimmer_pool' or self.pool_type == 'Diving_pool': 

255 a = 4.5 

256 n = 1 

257 m = None 

258 N = self.area * n/a 

259 elif self.pool_type == 'Nonswimmer_pool'\ 

260 or self.pool_type == 'Freeform_pool' \ 

261 or self.pool_type == 'Multipurpose_pool': 

262 a = 2.7 

263 n = 1 

264 m = None 

265 N = self.area * n / a 

266 elif self.pool_type == 'Child_pool': 

267 a = None 

268 n = None 

269 m = 2 

270 N = self.area * m * k 

271 else: 

272 raise ValueError(f"Unknown pool type: {self.pool_type}. Only the following pool types are supported: " \ 

273 f"'Swimmer_pool', 'Nonswimmer_pool', 'Multipurpose_pool', 'Child_pool'," \ 

274 f" 'Multipurpose_pool', 'Diving_pool', 'Freeform_pool'") 

275 

276 # V_H: hygienic minimum volume flow 

277 # V_K: hygienic minimum volume flow for children pool 

278 # V_B: hydraulic minimum volume flow 

279 V_H = N / k # Ratio of nominal load divided by factor k 

280 V_B = 1 * self.perimeter 

281 if self.pool_type == 'Child_pool' and self.area < 20: 

282 V_K = m * self.volume 

283 else: 

284 V_K = 0 

285 

286 # volume_flow in m3/h 

287 if self.perimeter > 40: 

288 self.volume_flow = max(V_B, V_H) / 3600 

289 elif V_K > 0: 

290 self.volume_flow = min(V_H, V_K, V_B) / 3600 

291 else: 

292 self.volume_flow = min(V_H, V_B) / 3600 

293 

294 # volume_flow_night 

295 if self.night_set_back_volume_flow is None: 

296 self.night_set_back_volume_flow = True 

297 

298 if self.night_set_back_volume_flow == True: 

299 if self.volume_flow_night is None: 

300 if V_B > 30: 

301 self.volume_flow_night = V_B/3600 

302 else: 

303 self.volume_flow_night = 30/3600 

304 else: 

305 self.volume_flow_night = self.volume_flow 

306 

307 # volume_storage 

308 if self.filter_type is None: 

309 self.filter_type = 'Open suction filter' 

310 if self.water_type is None: 

311 self.water_type = "Fresh water" 

312 # help parameter v_f 

313 if self.filter_type == 'Activated carbon filter with ozone' \ 

314 or self.filter_type == 'Two-layer filter with ozone': 

315 v_f = 50 

316 elif self.filter_type == 'Open quick filter': 

317 v_f = 15 

318 elif self.filter_type == 'Closed quick filter' \ 

319 or self.filter_type == 'Closed sorption filter' \ 

320 or self.filter_type == 'Open suction filter' \ 

321 or self.filter_type == 'Quantozone filter' \ 

322 or self.filter_type == 'Quartz gravel filter' \ 

323 or self.filter_type == 'Two-layer filter': 

324 if self.water_type == 'Fresh water': 

325 v_f = 30/3600 

326 else: 

327 v_f = 20/3600 

328 else: 

329 raise ValueError(f"Unknown filter type: {self.filter_type}. " 

330 f"Only the following filter types are supported: " 

331 f"'Activated carbon filter with ozone','Two-layer filter with ozone'," 

332 f"'Open quick filter','Closed quick filter','Closed sorption filter'," 

333 f"'Quantozone filter','Quartz gravel filter','Two-layer filter'") 

334 

335 # help parameter a_filter 

336 a_filter = self.volume_flow / v_f 

337 

338 # help parameter V_v 

339 if a is not None: 

340 V_v = 0.075 * self.area / a 

341 else: 

342 V_v = None 

343 

344 # help parameter V_w 

345 V_w = 0.052 * self.area * 10**(-0.144 * self.volume_flow/self.perimeter) 

346 

347 # help parameter V_fs 

348 V_fs = a_filter * 5 

349 

350 # Calculation of volume_storage 

351 self.volume_storage = V_v + V_w + V_fs 

352 

353 # Calculation of beta_in_use according to VDI 2089 in m/s 

354 if self.pool_type == 'Freeform_pool' or self.pool_type == 'Multipurpose_pool': 

355 self.beta_in_use = 40/3600 

356 else: 

357 if self.depth > 1.35: 

358 self.beta_in_use = 28/3600 

359 else: 

360 self.beta_in_use = 40/3600 

361 

362 # Caluclation of m_flow_waste_water 

363 if self.num_filter_rinses is None: 

364 self.num_filter_rinses = 2 

365 # avereage number of visitors KOK 

366 if self.num_visitors is None: 

367 self.num_visitors = round(self.area ** 0.58, 0) 

368 m_visitors = 0.03 * 995.65 * self.num_visitors * 1/(24*3600) 

369 m_filter = 995.65 * self.num_filter_rinses * V_fs * 1/(7*24*3600) 

370 self.m_flow_waste_water = max(m_visitors, m_filter) 

371 

372 # Calculation of evaporation mass flow at maximal occupancy according to VDI 2089 in kg/h 

373 R_D = 461.52 # specific gas constant for steam 

374 aritmethic_temperature = (self.temperature + self.temperature_air) /2 

375 # saturation pressure in Pa 

376 p_sat_air_temp = self.calc_sat_pressure(self.temperature_air) 

377 p_sat_pool_temp = self.calc_sat_pressure(self.temperature) 

378 

379 

380 self.m_flow_evap_pool_used = -((self.beta_in_use*3600)/(R_D*aritmethic_temperature))* \ 

381 (p_sat_pool_temp-p_sat_air_temp)*self.area 

382 print('self.m_flow_evap_pool_used',self.m_flow_evap_pool_used) 

383 

384 # heat recovery rinsing wastewater 

385 if self.use_heat_recovery is None: 

386 self.use_heat_recovery = True 

387 if self.eta_heat_recovery is None: 

388 self.eta_heat_recovery = 0.8 

389 if self.dp_heat_exchanger is None: 

390 self.dp_heat_exchanger = 350 

391 

392 # wastewater treatment 

393 if self.use_water_recycling is None: 

394 self.use_water_recycling = False 

395 if self.x_recycling is None: 

396 self.x_recycling = 0.8 

397 

398 

399 # pool cover 

400 if self.use_pool_cover is None: 

401 self.use_pool_cover = False 

402 

403 # wave pool 

404 if self.use_wave_pool is None: 

405 self.use_wave_pool = False 

406 if self.wave_period is None: 

407 self.wave_period = 1800 

408 if self.wave_height is None: 

409 self.wave_height = 0 

410 if self.wave_width is None: 

411 self.wave_width = 0 

412 if self.wave_period_start is None: 

413 self.wave_period_start = 600 

414 if self.wave_share_period is None: 

415 self.wave_share_period = 10/30*100 

416 

417 # pool wall construction 

418 if self.pool_wall_construction_type is None: 

419 self.pool_wall_construction_type = 'concrete_with_insulation'

422 def calc_sat_pressure(self, temp): 

423 log_pw = 10.79574 * (1.0 - 273.16 / temp) \ 

424 - 5.02800 * math.log10(temp / 273.16) \ 

425 + 1.50475E-4 * (1 - math.pow(10, (-8.2969 * (temp/ 273.16 - 1.0)))) + 0.42873E-3 * \ 

426 (math.pow(10, (+4.76955 * (1.0 - 273.16 / temp))) - 1) + 0.78614 

427 p_sat = math.pow(10, log_pw)*100 

428 return p_sat
