135 def calc_equivalent_res(self): 

136 """Equivalent resistance VDI 6007 

137 

138 Calculates the equivalent resistance of a wall according to VDI 

139 6007 guideline. 

140 

141 Parameters 

142 ---------- 

143 t_bt : int 

144 time constant according to VDI 6007 (default t_bt = 7) 

145 """ 

146 self.set_calc_default() 

147 number_of_layer, density, thermal_conduc, heat_capac, thickness = \ 

148 self.gather_element_properties() 

149 

150 r_layer = thickness / thermal_conduc 

151 c_layer = heat_capac * density * thickness # *1000 

152 

153 for layer_count in r_layer: 

154 self.r1 += layer_count / self.area 

155 

156 for layer_count in c_layer: 

157 self.c1 += layer_count
