1# # Example 1: Generate archetype buildings using TEASER API 

2# This module contains an example how to generate archetype buildings 

3# using TEASER API functions. 

4# You can run this example using the [jupyter-notebook](https://mybinder.org/v2/gh/RWTH-EBC/TEASER/main?labpath=docs%2Fjupyter_notebooks) 

5 

6 

7def example_generate_archetype(): 

8 """This function demonstrates the generation of residential and 

9 non-residential archetype buildings using the API function of TEASER""" 

10 

11 # First step: Import the TEASER API (called Project) into your Python 

12 # module 

13 

14 from teaser.project import Project 

15 

16 # To use the API, instantiate the Project class and rename the project. 

17 # Be careful: Dymola does not like whitespaces in names and filenames, thus we will delete them anyway in TEASER. 

18 

19 prj = Project() 

20 prj.name = "ArchetypeExample" 

21 

22 # There are two different types of archetype groups: residential and 

23 # non-residential buildings. Two API functions offer the opportunity to 

24 # generate specific archetypes. 

25 

26 # To generate residential archetype buildings the function 

27 # Project.add_residential() is used. Seven parameters are compulsory, 

28 # additional parameters can be set according to the used method. `construction_data` 

29 # and `geometry_data` are used to distinguish between different construction and archetype 

30 # methods. The name, year_of_construction, number and height of floors 

31 # and net_leased_area need to be set to provide enough information for 

32 # archetype generation. For specific information on the parameters please 

33 # read the docs. 

34 

35 prj.add_residential( 

36 construction_data='iwu_heavy', 

37 geometry_data='iwu_single_family_dwelling', 

38 name="ResidentialBuilding", 

39 year_of_construction=1988, 

40 number_of_floors=2, 

41 height_of_floors=3.2, 

42 net_leased_area=200.0) 

43 

44 # To generate non-residential archetype buildings (in this case an 

45 # office and a laboratory (a.k.a. institute)) the function 

46 # Project.add_residential() is used. The meaning of compulsory parameters 

47 # does not differ from the residential archetype building. 

48 

49 prj.add_non_residential( 

50 construction_data='iwu_heavy', 

51 geometry_data='bmvbs_office', 

52 name="OfficeBuilding", 

53 year_of_construction=1988, 

54 number_of_floors=4, 

55 height_of_floors=3.5, 

56 net_leased_area=4500.0) 

57 

58 prj.add_non_residential( 

59 construction_data='iwu_heavy', 

60 geometry_data='bmvbs_institute', 

61 name="InstituteBuilding", 

62 year_of_construction=1952, 

63 number_of_floors=5, 

64 height_of_floors=4.0, 

65 net_leased_area=3400.0) 

66 

67 prj.add_non_residential( 

68 construction_data='iwu_heavy', 

69 geometry_data='bmvbs_institute', 

70 name="InstituteBuildingMoisture", 

71 year_of_construction=1980, 

72 number_of_floors=3, 

73 height_of_floors=4.2, 

74 net_leased_area=3600.0, 

75 internal_gains_mode=3) 

76 

77 # Besides `iwu` and `bmvbs` there is a third option for archetype 

78 # generation. We integrated the typology of TABULA Germany 

79 # (http://webtool.building-typology.eu/#bm) and other countries are about to 

80 # follow. To use TABULA archetype simple choose the default `tabula_de_standard` as the construction_data 

81 # and `tabula_de_single_family_house`, `tabula_de_multi_family_house`, `tabula_de_terraced_house` or 

82 # `tabula_de_apartment_block` as the geometry_data. In addition you can specify the 

83 # construction type of TABULA, chose between `tabula_de_standard` (default), 

84 # `tabula_de_retrofit` or `tabula_de_adv_retrofit`. In this case we generate one 

85 # single and one multi family house with TABULA typology. 

86 

87 # Please not: as we need to load the construction information which are 

88 # rather big for TABULA, switching from one typology to another in the same 

89 # Project takes some seconds. If you know from beginning you will only use 

90 # TABULA typology you should instantiate you Project class without loading 

91 # data. Project(load_data=False). 

92 

93 prj.add_residential( 

94 construction_data='tabula_de_standard', 

95 geometry_data='tabula_de_single_family_house', 

96 name="ResidentialBuildingTabula", 

97 year_of_construction=1988, 

98 number_of_floors=3, 

99 height_of_floors=3.2, 

100 net_leased_area=280.0) 

101 

102 prj.add_residential( 

103 construction_data='tabula_de_retrofit', 

104 geometry_data='tabula_de_multi_family_house', 

105 name="ResidentialBuildingTabulaMulti", 

106 year_of_construction=1960, 

107 number_of_floors=4, 

108 height_of_floors=3.2, 

109 net_leased_area=600.0) 

110 

111 return prj 

112 

113 

114if __name__ == '__main__': 

115 prj = example_generate_archetype() 

116 

117 print("Example 1: That's it! :)")
